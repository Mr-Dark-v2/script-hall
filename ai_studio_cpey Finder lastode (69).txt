local function addHoverEffect(button)

    outline.Color = Color3.fromRGB(255, 255, 255)
    outline.Thickness = 1
    outline.Transparency = 1
    outline.Parent = button
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 26, 0, 26)
        }):Play()
        TweenService:Create(outline, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Transparency = 0.3
        }):Play()
    end)
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 24, 0, 24)
        }):Play()
        TweenService:Create(outline, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Transparency = 1
        }):Play()
    end)
end

-- Mutation Color Map
local function getMutationColor(mutation)
    local colorMap = {
        Diamond = Color3.fromRGB(0, 220, 255),
        Gold = Color3.fromRGB(255, 215, 0),
        Mythic = Color3.fromRGB(255, 105, 180),
        Epic = Color3.fromRGB(180, 0, 255),
        Rare = Color3.fromRGB(0, 180, 255),
        Common = Color3.fromRGB(150, 150, 150),
    }
    return colorMap[mutation] or Color3.fromRGB(200, 200, 200)
end

local function createSecretRarityEffect(parentTextLabel)
    local player = game.Players.LocalPlayer
    local playerGui = player:WaitForChild("PlayerGui")

    local effectScreenGui = Instance.new("ScreenGui")
    effectScreenGui.Name = "TextEffectGui_" .. tostring(tick()) -- Unique name
    effectScreenGui.Parent = playerGui
    effectScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Global -- Allows it to draw over other things

    -- ‚≠ê MODIFIED: Even smaller container
    local containerWidth = 42
    local containerHeight = 26
    local textContainer = Instance.new("Frame")
    textContainer.Parent = effectScreenGui
    textContainer.Size = UDim2.fromOffset(containerWidth, containerHeight)
    textContainer.BackgroundTransparency = 1
    textContainer.ClipsDescendants = true

    local baseText = Instance.new("TextLabel")
    baseText.Parent = textContainer
    baseText.Size = UDim2.fromScale(1, 1)
    baseText.BackgroundTransparency = 1
    baseText.Text = "secret"
    baseText.TextColor3 = Color3.new(1, 1, 1)
    baseText.TextSize = 14 -- ‚≠ê MODIFIED: Smaller text
    baseText.Font = Enum.Font.GothamBold
    baseText.TextXAlignment = Enum.TextXAlignment.Left
    baseText.TextYAlignment = Enum.TextYAlignment.Top
    baseText.ZIndex = 1

    local baseStroke = Instance.new("UIStroke")
    baseStroke.Parent = baseText
    baseStroke.Color = Color3.new(0, 0, 0)
    baseStroke.Thickness = 0.7 -- ‚≠ê MODIFIED: Thinner outline

    local SCROLL_SPEED = 0.25
    local TRAIN_LENGTH = 8
    local OVERLAP_BUFFER = 0.4
    local VISIBLE_HEIGHT = 1.0
    local TOTAL_HEIGHT = VISIBLE_HEIGHT + OVERLAP_BUFFER * 2

    local shineTemplate = Instance.new("UIGradient")
    shineTemplate.Rotation = 90
    shineTemplate.Transparency = NumberSequence.new({
        NumberSequenceKeypoint.new(0, 1),
        NumberSequenceKeypoint.new(0.39, 1),
        NumberSequenceKeypoint.new(0.4, 0),
        NumberSequenceKeypoint.new(0.6, 0),
        NumberSequenceKeypoint.new(0.61, 1),
        NumberSequenceKeypoint.new(1, 1)
    })

    local textLayers = {
        { color = Color3.new(0, 0, 0), outline = Color3.new(1, 1, 1) },
        { color = Color3.new(1, 1, 1), outline = Color3.new(0, 0, 0) }
    }
    local shines = {}
    local spacing = TOTAL_HEIGHT / TRAIN_LENGTH
    for i = 1, TRAIN_LENGTH do
        local layerInfo = textLayers[(i - 1) % 2 + 1]
        local overlay = baseText:Clone()
        overlay.Parent = textContainer
        overlay.TextColor3 = layerInfo.color
        overlay.ZIndex = 10 + i
        overlay:FindFirstChild("UIStroke").Color = layerInfo.outline
        local shine = shineTemplate:Clone()
        shine.Parent = overlay
        shine.Enabled = true
        local startOffset = -OVERLAP_BUFFER - (spacing * (i - 1))
        shine.Offset = Vector2.new(0, startOffset)
        table.insert(shines, {
            shine = shine,
            speed = SCROLL_SPEED,
            overlap = OVERLAP_BUFFER,
            totalHeight = TOTAL_HEIGHT
        })
    end

    local heartbeat, renderConnection

    heartbeat = RunService.Heartbeat:Connect(function(deltaTime)
        for _, data in ipairs(shines) do
            local shine = data.shine
            local current = shine.Offset.Y
            local newY = current + data.speed * deltaTime
            if newY >= data.overlap then
                newY = newY - data.totalHeight
            end
            shine.Offset = Vector2.new(0, newY)
        end
    end)

    local function updatePosition()
        if not parentTextLabel or not parentTextLabel.Parent then
            if heartbeat then heartbeat:Disconnect() end
            if renderConnection then renderConnection:Disconnect() end
            effectScreenGui:Destroy()
            return
        end
        local absPos = parentTextLabel.AbsolutePosition
        local absSize = parentTextLabel.AbsoluteSize
        -- ‚≠ê MODIFIED: Adjusted positioning for new size
        textContainer.Position = UDim2.fromOffset(absPos.X + absSize.X + 5, absPos.Y + 2)
    end
    renderConnection = RunService.RenderStepped:Connect(updatePosition)

    parentTextLabel.AncestryChanged:Connect(function(_, newParent)
        if not newParent then
            if heartbeat then heartbeat:Disconnect() end
            if renderConnection then renderConnection:Disconnect() end
            effectScreenGui:Destroy()
        end
    end)

    updatePosition()
end

            TweenService:Create(joinOutline, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Transparency = 0.3 }):Play()
        end
    end)

    joinButton.MouseLeave:Connect(function()
        if joinButton.Visible then
            TweenService:Create(joinButton, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Size = UDim2.new(0, 75, 0, 24) }):Play()
            TweenService:Create(joinOutline, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Transparency = 1 }):Play()
        end
    end)

    joinButton.MouseButton1Click:Connect(function()
        if joinButton.Visible then
            local expandTween = TweenService:Create(joinButton, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Size = UDim2.new(0, 77, 0, 24.8) })
            expandTween:Play()
            expandTween.Completed:Connect(function()
                TweenService:Create(joinButton, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Size = UDim2.new(0, 75, 0, 24) }):Play()
            end)
        end
        print("Joining server for:", petName)
    end)

    local inputCatcher = Instance.new("TextButton")
    inputCatcher.Name = "InputCatcher"
    inputCatcher.Size = UDim2.new(1, 0, 1, 0)
    inputCatcher.BackgroundTransparency = 1
    inputCatcher.Text = ""
    inputCatcher.ZIndex = 5
    inputCatcher.Parent = card

    local tweenInfo = TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)

    local function enterHover()
        TweenService:Create(card, tweenInfo, { Size = hoverSize, Position = hoverPosition, ZIndex = 2 }):Play()
        TweenService:Create(cardOutline, tweenInfo, { Thickness = 1.5 }):Play()
        TweenService:Create(shadow, tweenInfo, { Position = UDim2.new(-0.02, 0, -0.03, 0), ImageTransparency = 0.8 }):Play()
        TweenService:Create(mutationLabel, tweenInfo, { TextTransparency = 0.2 }):Play()
        TweenService:Create(statLabel, tweenInfo, { TextTransparency = 0.1 }):Play()
    end

    local function enterSelectedState()
        TweenService:Create(card, tweenInfo, { Size = expandedSize, Position = hoverPosition, ZIndex = 3 }):Play()
        TweenService:Create(cardOutline, tweenInfo, { Thickness = 1.5 }):Play()
        TweenService:Create(shadow, tweenInfo, { Position = UDim2.new(-0.02, 0, -0.03, 0), ImageTransparency = 0.8 }):Play()
        TweenService:Create(mutationLabel, tweenInfo, { TextTransparency = 0.2 }):Play()
        TweenService:Create(statLabel, tweenInfo, { TextTransparency = 0.1 }):Play()
        joinButton.Visible = true
        TweenService:Create(joinButton, tweenInfo, { BackgroundTransparency = 0, TextTransparency = 0 }):Play()
    end

    local function leaveState()
        TweenService:Create(card, tweenInfo, { Size = restingSize, Position = restingPosition, ZIndex = 1 }):Play()
        TweenService:Create(cardOutline, tweenInfo, { Thickness = 0 }):Play()
        TweenService:Create(shadow, tweenInfo, { Position = UDim2.new(-0.02, 0, -0.02, 0), ImageTransparency = 0.85 }):Play()
        TweenService:Create(mutationLabel, tweenInfo, { TextTransparency = 0.6 }):Play()
        TweenService:Create(statLabel, tweenInfo, { TextTransparency = 0.5 }):Play()
        local fadeOutTween = TweenService:Create(joinButton, tweenInfo, { BackgroundTransparency = 1, TextTransparency = 1 })
        fadeOutTween:Play()
        fadeOutTween.Completed:Connect(function() joinButton.Visible = false end)
    end

    inputCatcher.MouseEnter:Connect(function() if selectedCard ~= card then enterHover() end pcall(function() UserInputService.MouseIcon = "rbxasset://textures/MousePointingHand.png" end) end)
    inputCatcher.MouseLeave:Connect(function() if selectedCard ~= card then leaveState() end pcall(function() UserInputService.MouseIcon = nil end) end)
    inputCatcher.MouseButton1Click:Connect(function()
        local cardWasAlreadySelected = (selectedCard == card)
        if deselectCurrentCardFunction then deselectCurrentCardFunction() deselectCurrentCardFunction = nil selectedCard = nil end
        if not cardWasAlreadySelected then enterSelectedState() selectedCard = card deselectCurrentCardFunction = leaveState end
    end)
    return card
end

-- Scrolling Area Function
local function createScrollingArea(parent, xPos)
    local frame = Instance.new("ScrollingFrame")
    frame.Size = UDim2.new(0.47, 0, 1, 0)
    frame.Position = UDim2.new(xPos, xPos == 0 and 5 or 0, 0, 0)
    frame.BackgroundTransparency = 1
    frame.BorderSizePixel = 0
    frame.ScrollBarThickness = 4
    frame.ScrollBarImageColor3 = Color3.fromRGB(80, 80, 85)
    frame.CanvasSize = UDim2.new(0, 0, 0, 0)
    frame.Parent = parent
    frame.ClipsDescendants = false

    local scrollBarCorner = Instance.new("UICorner")
    scrollBarCorner.CornerRadius = UDim.new(0, 2)
    scrollBarCorner.Parent = frame

    local scrollPadding = Instance.new("UIPadding")
    scrollPadding.PaddingLeft = UDim.new(0, 12)
    scrollPadding.PaddingRight = UDim.new(0, 24)
    scrollPadding.PaddingTop = UDim.new(0, 60)
    scrollPadding.PaddingBottom = UDim.new(0, 20)
    scrollPadding.Parent = frame

    local layout = Instance.new("UIListLayout")
    layout.SortOrder = Enum.SortOrder.LayoutOrder
    layout.Padding = UDim.new(0, 10)
    layout.Parent = frame

    return frame, layout
end
